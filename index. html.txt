  <script>
    // Auto confetti on page load
    window.onload = () => {
      launchConfetti();
    };

    // Confetti function
    function launchConfetti() {
      const duration = 3000;
      const end = Date.now() + duration;

      const colors = ['#ff4f81', '#ffffff', '#ffd700', '#ff85a2'];

      (function frame() {
        confetti({
          particleCount: 5,
          angle: 60,
          spread: 55,
          origin: { x: 0 },
          colors: colors
        });

        confetti({
          particleCount: 5,
          angle: 120,
          spread: 55,
          origin: { x: 1 },
          colors: colors
        });

        if (Date.now() < end) {
          requestAnimationFrame(frame);
        }
      })();
    }
  </script>
